Write a Python program that takes a number as input and prints its square root using Newton's method and a while loop.

            def square_root(n):
            guess = n / 2
            while abs(guess * guess - n) > 1e-6:
                guess = 0.5 * (guess + n / guess)
            return guess
        
        num = float(input("Enter a positive number: "))
        if num >= 0:
            print(f"Square root of {num} is approximately {square_root(num):.6f}")
        else:
            print("Please enter a positive number.")

            original = n
            reversed_num = 0
            while n > 0:
                digit = n % 10
                reversed_num = reversed_num * 10 + digit
                n //= 10
            return original == reversed_num
