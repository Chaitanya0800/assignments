Basic Iterator: Write an iterator class that iterates over a given list of integers and returns only the even numbers.

            class EvenNumberIterator:
            def __init__(self, numbers):
                self.numbers = numbers
                self.index = 0
        
            def __iter__(self):
                return self
        
            def __next__(self):
                while self.index < len(self.numbers):
                    current_number = self.numbers[self.index]
                    self.index += 1
                    if current_number % 2 == 0:
                        return current_number
                raise StopIteration

Custom Range Iterator: Create a custom range iterator class that mimics the behavior of Pythonâ€™s built-in range() function but allows for both positive and negative steps.

            class CustomRangeIterator:
            def __init__(self, start, stop, step=1):
                self.current = start
                self.stop = stop
                self.step = step
        
            def __iter__(self):
                return self
        
            def __next__(self):
                if (self.step > 0 and self.current >= self.stop) or (self.step < 0 and self.current <= self.stop):
                    raise StopIteration
                current_value = self.current
                self.current += self.step
                return current_value

Question 15:

Retry Decorator: Implement a decorator that retries a function a specified number of times if it raises an exception..

            import random

            def retry(max_attempts):
                def decorator(func):
                    def wrapper(*args, **kwargs):
                        for attempt in range(max_attempts):
                            try:
                                return func(*args, **kwargs)
                            except Exception as e:
                                print(f"Attempt {attempt + 1} failed: {e}")
                                if attempt < max_attempts - 1:
                                    wait_time = random.uniform(0.5, 1.0)
                                    print(f"Retrying in {wait_time:.2f} seconds...")
                                    time.sleep(wait_time)
                        raise Exception(f"Function {func.__name__} failed after {max_attempts} attempts")
                    return wrapper
                return decorator

Timezone Conversion: Create a script that converts a given UTC time to a specified timezone (e.g., converting UTC to EST).

            import datetime
            import pytz
            
            def convert_utc_to_timezone(utc_time, target_timezone):
                utc = pytz.utc
                target_tz = pytz.timezone(target_timezone)
                utc_dt = utc.localize(utc_time)
                target_dt = utc_dt.astimezone(target_tz)
                return target_dt
