Create a Python program that utilizes multi-line comments to describe the functionality of a function or a section of code.

# This program demonstrates the use of multi-line comments in Python to describe the functionality of a function
'''
This function calculates the factorial of a given number.
It uses recursion to calculate the factorial.
Input: n (integer)
Output: Factorial of n
'''
def factorial(n):
    if n == 0:
        return 1
    else:
        return n * factorial(n - 1)

# Test the factorial function with the number 5
result = factorial(5)
print(f"The factorial of 5 is: {result}").

Question 16:

Write a Python program that demonstrates the use of the nonlocal keyword to modify variables from an outer function within a nested function.

# Outer function
def outer_function():
    outer_var = 10

    # Nested function that modifies the outer variable using nonlocal keyword
    def nested_function():
        nonlocal outer_var
        outer_var = 20
        print("Inside nested function - Modified outer variable:", outer_var)

    # Call the nested function
    nested_function()

    print("Inside outer function - Outer variable after modification:", outer_var)

# Call the outer function
outer_function()
